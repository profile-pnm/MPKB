// Установка и настройка тайлинга i3wm (на примере Arch)

	== Установка необходимых компонентов ==
1. $ sudo pacman -S xorg-xinit xorg-server(установкаоконной системы xorg)
2. $ sudo pacman -S i3 (установить i3wm, i3status, dmenu alacritty)
3. $ mkdir vim .xinitrc (в него записать следующее: exec i3)
4. $ sudo pacman -S xf86-video-intel (дрова для видеокарты Intel); xf86-video-nouveau (для NVIDIA); xf86-video-ati (AMD)
	 * Для виртульной машины ( $ sudo pacman -S xf86-video-vesa )
5. $ sudo reboot

	== Layout на i3wm ==
1. sudo setxkbmap -option grp:alt_shift_toggle “us, ru” (ставим русско-английскую раскладку)
2. Переходим в домашний конфиг с i3. Записываем настройку для автозапуска setxkbmap: 
	| #Layoutexec 
	| setxkbmap -option 'grp:alt_shift_toggle'
		или (в том случае если не будет работать)	
	| exec "setxkbmap -model pc104 -layout us,ru -variant ,, -option 'grp:alt_shift_toggle'"

# resize window (you can also use the mouse for that)
mode "resize" {
        # These bindings trigger as soon as you enter the resize mode

        # Pressing left will shrink the window’s width.
        # Pressing right will grow the window’s width.
        # Pressing up will shrink the window’s height.
        # Pressing down will grow the window’s height.
        bindsym h resize shrink width 10 px or 10 ppt
        bindsym j resize grow height 10 px or 10 ppt
        bindsym k resize shrink height 10 px or 10 ppt
        bindsym l resize grow width 10 px or 10 ppt

      # same bindings, but for the arrow keys
        bindsym Left resize shrink width 10 px or 10 ppt
        bindsym Down resize grow height 10 px or 10 ppt
        bindsym Up resize shrink height 10 px or 10 ppt
        bindsym Right resize grow width 10 px or 10 ppt

        # back to normal: Enter or Escape
        bindsym Return mode "default"
        bindsym Escape mode "default"
}

bindsym $mod+r mode "resize"
