// Метод с tar:

1. СОЗДАНИЕ БЭКАПА:
<< В LiveUSB переключаемся на root >>
$ sudo su

<< Создадим временно для флешки >>
# mkdir /media

<< Монтируем раздел, который будем архивировать >>
# mount -o ro /dev/нужный_раздел"sda" /mnt

<< Примонтируем носитель для чтения-записи, чтобы писать туда бэкап >>
# mount -o rw /dev/флешка /media

<< Создаем архив >>
tar -cvzpf /media/backupDDMMYY.tgz --exclude={"/mnt/dev/*","/mnt/sys/*","/mnt/proc/*","/mnt/tmp/*","/mnt/run/*","/mnt/lost+found/*","/mnt/var/cache/pacman/pkg/*"} /mnt 

<< Отмонтируем диск и флешку >>
# umount /mnt
# umount /media

<< Перезагружаемся >>
# reboot

=============================================================================

2. ВОССТАВНОВЛЕНИЕ БЭКАПА:
<< Грузимся с LiveUSB и переключаемся в root >>
...

<< Делаем разметку диска, на который будет стоять система >>
# cfdisk <под систему "/" должен быть тип 83Linux, если нет других разделов, то
помечаем системный раздел загрузочным и сохраняем изменения>

<< Форматируем файловую систему "/" >>
# mkfs.ext4 /dev/корень


2:1. ЭТАП РАСПАКОВКИ АРХИВА В СВЕЖУЮ :
<< Монтируем отформатированный раздел >>
# mount /dev/корень /mnt

<< Монтируем флешку >>
# mkdir /media
# mount -o rw /dev/флешка /media


<< Распаковывам архив с бэкапом с носителя >>
# tar --same-owner -xvpf /media/backupDDMMYY.tgz -C /mnt
// параметр --same-owner сохраняет инфу о владельцах у распаковываемых файлов,
p - сохраняет права доступа, f - указывает файл, который распаковываем

<< Создать каталоги, которые были исключены >>
.... # mkdir /sys && mkdir /proc && mkdir /dev && mkdir /tmp && mkdir /run && mkdir /lost+found ....

<< Создаем загрузочную запись >>
# mount --bind /dev /mnt/dev
# mount --bind /proc /mnt/proc 
# mount --bind /sys /mnt/sys
// параметр bind нужен для того, чтобы те каталоги, которые используются в live-режиме
были доступны сразу в двух местах и в liveUSB и в будущей системе

<< Переключаемся на новую систему использую chroot >>
# chroot /mnt

<< Устанавливаем grub2 >>
# grub-install /dev/наш_диск

<< Обновляем инфу из fstab >>
# update-grub

<< Для работы GRUB надо указать ему верные UUID разделов в fstab >>
<< Сейчас там прописаны разделы пред. системы и поэтому нам вызвать софт для просмотра тек. UUID >>
# blkid

<< Переписываем в /etc/fstab верные UUID и сохраняем изменения >>
...

<< Возвращаемся в Live-систему >>
# exit

<< Размонтируем все каталоги >>
# umount /mnt/dev
# umount /mnt/proc
# umount /mnt/sys
# umount /mnt

<< Перезапуск >>
# reboot

// Метод с rsync:

1. СОЗДАНИЕ БЭКАПА

<< На примонтированную флешку создаем каталог >>
$ sudo mkdir /mnt/флешка/названиеКаталогаБэкапа

<< Выйдем в корень пользователем root >>
# cd /

<< Начнем пробное резервное копирование >>
# rsync -aAXv --delete --dry-run --exclude={"/dev/*","/sys/*","/proc/*","/tmp/*,"/run/*","/lost+found","/var/lib/dhcpcd/*","/mnt/*"} / /mnt/флешка/названиеКаталогаБэкапа

<< Если ошибок нет, то делаем реальный бэкап, только без ключа --dry-run >>
# rsync -aAXv --delete --exclude={"/dev/*","/sys/*","/proc/*","/tmp/*,"/run/*","/lost+found","/var/lib/dhcpcd/*","/mnt/*"} / /mnt/флешка/названиеКаталогаБэкапа

=================================================
2. Восстановление бэкапа в Live-USB / Live-CD

<< Создадим два каталога для монтирования флешки и системы >>
# mkdir /mnt/system /mnt/usb

<< Смонтируем бэкап на флешке в созданный каталог и систему, которую будем восстанавливать >>
# mount /dev/раздел_коренного_диска /mnt/system 
# mount /dev/флешка /mnt/usb

<< Приступим к самому восстановлению >>
# rsync -aAXv --delete --exclude="lost+found" /mnt/usb/ /mnt/system/

<< После успешного восстановление отмнотируем диск и флешку >>
# umount /dev/раздел_коренного_диска /mnt/system
# umount /dev/флешка /mnt/usb

<< Перезагружаем систему >>
# reboot
